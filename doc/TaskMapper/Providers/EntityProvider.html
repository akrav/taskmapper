def raise_implementation_instructions(method, args)
  def args.to_s
    empty? ? "" : "(#{map(&:class).join ','})"
  end
  
  raise TaskMapper::Exceptions::ImplementationNotFound
    .new(factory.provider_module, entity, method, args)
end
